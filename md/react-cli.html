react-cli

创建一个react应用

npm install -g create-react-app

创建应用

react-create-app first-project

创建完成之后我们进入到目录运行

npm run start

查看工具命令可以通过查看package.json文件中的script

添加react-router进行一个spa的开发

npm install --save react-router-dom

拆分路由可以用react-loadable

npm install react-loadable --save

import React, { Component } from 'react'
import { BrowserRouter as Router, Route, Switch,Link } from 'react-router-dom'
import Loadable from 'react-loadable'
import Loading from './loading';


const NoFound = Loadable({
  loader: () => import(/* webpackChunkName:"AAA" */'./components/AAA'),
  loading: Loading
})

class App extends Component {
  render() {
    return (
      <Router>
        <Switch>
          <Route exact path="/" component={() => <h2><p><Link to="/home">home</Link></p><p><Link to="/about">About</Link></p></h2>} />

          <Route exact path="/home" component={() => <h2>Home</h2>} />
          <Route exact path="/about" component={() => <h2>about</h2>} />
          <Route component={NoFound} />
        </Switch>
      </Router>
    )
  }
}
export default App
