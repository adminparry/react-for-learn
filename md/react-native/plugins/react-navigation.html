createStackNavigator

创建一个路由栈

import React from "react";
import { View, Text } from "react-native";
import { createStackNavigator, createAppContainer } from "react-navigation";

class HomeScreen extends React.Component {
  render() {
    return (
      <View style={{ flex: 1, alignItems: "center", justifyContent: "center" }}>
        <Text>Home Screen</Text>
      </View>
    );
  }
}

const AppNavigator = createStackNavigator({
  Home: {
    screen: HomeScreen
  }
});

export default createAppContainer(AppNavigator);

createAppContainer

包装成组件返回

const AppContainer = createAppContainer(AppNavigator);

export default class App extends React.Component {
  render() {
    return <AppContainer />;
  }
}

this.props.navigation.navigate

路由跳转

navigate 跳转到定义的路由栈 如果名字填写错误则不进行跳转

push 作为一个新的栈添加

goBack 关闭当前返回到上一个操作的路由栈

popToTop 返回到第一个路由栈

addListener - navigation生命周期监听

isFocused - 是否是当前路由

state - 路由组件的state

setParams - 设置路由参数

getParam - 获取路由参数

dispatch - 发送action给router

dangerouslyGetParent - 返回父级navigation


import React from 'react';
import { Button, View, Text } from 'react-native';
import { createStackNavigator, createAppContainer } from 'react-navigation';

class HomeScreen extends React.Component {
  render() {
    return (
      <View style={{ flex: 1, alignItems: 'center', justifyContent: 'center' }}>
        <Text>Home Screen</Text>
        <Button
          title="Go to Details"
          onPress={() => this.props.navigation.navigate('Details')}
        />
      </View>
    );
  }
}

Navigation的生命周期

willFocus, willBlur, didFocus and didBlur

传递参数

<!-- 通过设置第二个参数进行传递参数 -->
this.props.navigation.navigate('RouteName', { /* params go here */ })
<!-- 通过getParam获取参数如果没获取到则使用默认值第二个参数 -->
this.props.navigation.getParam(paramName, defaultValue)

header bar 标题

class HomeScreen extends React.Component {
  static navigationOptions = {
    title: 'Home',
  };

  /* render function, etc */
}

<!-- 也可以接受一个函数 -->
class DetailsScreen extends React.Component {
  static navigationOptions = ({ navigation, navigationOptions }) => {
    return {
      title: navigation.getParam('otherParam', 'A Nested Details Screen'),
    };
  };

  /* render function, etc */
}
