前期准备需要 nodejs xcode react-native-cli

<!-- 安装nodejs -->
brew install nodejs
npm install -g react-native-cli
react-native init demo
<!-- 打开xcode -->
cd demo && open ios/demo.xcodeproj
<!-- 运行react-native 然后点击xcode的build-->
npm run start
